webhook-engine / src / utils / hmac.js 
'use strict';

const crypto = require('crypto');
const SIGNING_KEY = process.env.WEBHOOK_SECRET || 'fallback_dev_secret';

/**
 * Signs a payload string using HMAC-SHA256.
 */
function signPayload(payload) {
  return crypto
    .createHmac('sha256', SIGNING_KEY)
    .update(payload)
    .digest('hex');
}

/**
 * Verifies a signature using timingSafeEqual.
 */
function verifySignature(payload, signature) {
  const expected = signPayload(payload);
  try {
    return crypto.timingSafeEqual(
      Buffer.from(expected, 'hex'),
      Buffer.from(signature, 'hex')
    );
  } catch {
    return false;
  }
}

module.exports = { signPayload, verifySignature };

webhook-engine / src / routes / events.js 
const express = require('express');
const { v4: uuidv4 } = require('uuid');
const { createEvent, deliverEvent } = require('../db/store');

const router = express.Router();

// POST /events — Accepts a new event and queues it for delivery
router.post('/', async (req, res) => {
  const { type, payload, webhook_url } = req.body;

  if (!type || typeof type !== 'string') {
    return res.status(400).json({ error: 'type is required' });
  }
  if (!webhook_url) {
    return res.status(400).json({ error: 'webhook_url is required' });
  }

  try {
    new URL(webhook_url);
  } catch {
    return res.status(400).json({ error: 'invalid URL' });
  }

  const event = createEvent({
    id: uuidv4(),
    type, payload, webhook_url,
    status: 'pending',
    created_at: new Date().toISOString(),
  });

  // Fire and forget — delivery runs in background
  setImmediate(() => deliverEvent(event));
  return res.status(201).json(event);
});

module.exports = router;
